#include <stdio.h>
#include <stdlib.h>

int main () {
  int *elementos;
  int n;

  printf("Quantos elementos vocÃª quer? ");
  scanf("%d", &n);

  // malloc -> memory allocation
  elementos = (int *) malloc(n * sizeof(int));

  if (!elementos) {
    printf("Not possible to create memory for this array\n");
    return 1;
  }

  for (int i = 0; i < n; i++) {
    elementos[i] = i + 1;
  }

  for (int i = 0; i < n; i++) {
    printf("%d\n", elementos[i]);
  }

  free(elementos);

  return 0;
}
